30      public Entanglement(Location where) {
31          this.timeline = new Timeline();
32          this.keyFrame = constructKeyFrame();
33          this.deltaX = 0;
34          this.deltaY = 0;
35          this.object = new Circle(5);
36          this.destination = where;
37          setPosition();
38      }
39  
40      @Override
41      public void fire() {
42          timeline.setCycleCount(1000);
43          timeline.getKeyFrames().add(keyFrame);
44          timeline.play();
45      }
46  
47      @Override
48      public void destroy() {
49          timeline.stop();
50          Pane pane = SyndromeFactory.getWorld().getGamePane();
51          if(!pane.getChildren().isEmpty()) {
52              pane.getChildren().remove(object);
53          }
54          SyndromeFactory.getWorld().removeProjectile(this);
55      }
56      
57      @Override
58      public Shape getObject() {
59          return object;
60      }

96      private void setPosition() {
97          Circle obj = (Circle) object;
98          obj.setTranslateX(destination.getX());
99          obj.setTranslateY(destination.getY());
100     }
101 
102     @Override
103     public void updateTranslation(Axis axis) {
104         Circle obj = (Circle) object;
105         Direction direction = SyndromeFactory.getWorld().getPlayer().getDirection();
106         int[] deltas = SyndromeFactory.getToolbox().directionToDelta(direction);
107         obj.setTranslateX(obj.getTranslateX() - deltas[0]);
108         obj.setTranslateY(obj.getTranslateY() - deltas[1]);
109     }
