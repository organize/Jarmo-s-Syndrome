1   package syndrome.logic.projectile.impl;
2   
3   import java.util.Random;
4   import javafx.animation.KeyFrame;
5   import javafx.animation.Timeline;
6   import javafx.event.ActionEvent;
7   import javafx.scene.layout.Pane;
8   import javafx.scene.shape.Circle;
9   import javafx.scene.shape.Shape;
10  import javafx.util.Duration;
11  import syndrome.logic.map.Location;
12  import syndrome.logic.projectile.Projectile;
13  import syndrome.other.SyndromeFactory;
14  
15  public class Entanglement implements Projectile {
16      
17      private final Timeline timeline;
18      private final KeyFrame keyFrame;
19      private final Shape object;
20      private double deltaX, deltaY;
21      
22      public Entanglement() {
23          this.timeline = new Timeline();
24          this.keyFrame = constructKeyFrame();
25          this.deltaX = 0;
26          this.deltaY = 0;
27          this.object = new Circle(15, 15, 5);
28      }
29  
30      @Override
31      public void fire() {
32          timeline.setCycleCount(1000);
33          timeline.getKeyFrames().add(keyFrame);
34          timeline.play();
35      }
36  
37      @Override
38      public void destroy() {
39          timeline.stop();
40          Pane pane = SyndromeFactory.getWorld().getGamePane();
41          if(!pane.getChildren().isEmpty()) {
42              pane.getChildren().remove(object);
43          }
44      }
45      
46      @Override
47      public Shape getObject() {
48          return object;
49      }
50  
51      private KeyFrame constructKeyFrame() {
52          return new KeyFrame(Duration.seconds(0.010), (ActionEvent event) -> {
53              Location destination = SyndromeFactory.getWorld().getPlayer().getLocation();
54              Circle obj = (Circle) object;
55              Random random = new Random();
56              double delta = random.nextInt(3) + 0.2;
57              if(destination.getX() != obj.getCenterX()
58                      && destination.getY() != obj.getCenterY()) {
59                  if(destination.getX() > obj.getCenterX()) {
60                      deltaX = delta;
61                  }
62                  if(destination.getX() < obj.getCenterX()) {
63                      deltaX = -delta;
64                  }
65                  if(destination.getY() > obj.getCenterY()) {
66                      deltaY = delta;
67                  }
68                  if(destination.getY() < obj.getCenterY()) {
69                      deltaY = -delta;
70                  }
71                  obj.setCenterX(obj.getCenterX() + deltaX);
72                  obj.setCenterY(obj.getCenterY() + deltaY);
73              }
74              timeline.setOnFinished((ActionEvent subHandler) -> {
75                  this.destroy();
76              });
77          });
78      }
79  
80      public Timeline getTimeline() {
81          return timeline;
82      }
83  }
