1   package syndrome.logic.map;
2   
3   import java.awt.geom.Point2D;
4   import java.util.Random;
5   import javafx.scene.input.MouseEvent;
6   import syndrome.entity.Player;
7   import syndrome.other.SyndromeFactory;
8   
9   /**
10   * Represents a point in the game screen.
11   * 
12   * @author Axel Wallin
13   */
14  public class Location {
15      
16      private double x, y;
17      
18      /**
19       * Creates a new location with the coordinates [0, 0].
20       */
21      public Location() {
22          this(0.0D, 0.0D);
23      }
24      
25      /**
26       * Creates a new location with the supplied coordinates.
27       * 
28       * @param x the x coordinate.
29       * @param y the y coordinate.
30       */
31      public Location(double x, double y) {
32          this.x = x;
33          this.y = y;
34      }
35      
36      public double getX() {
37          return x;
38      }
39      
40      public double getY() {
41          return y;
42      }
43      
44      /**
45       * Finds the distance between two Location instances.
46       * 
47       * @param location the location that needs to be checked.
48       * @return distance the distance between this and <i>location</i>.
49       */
50      public double distanceTo(Location location) {
51          return Point2D.distance(x, y, location.getX(), location.getY());
52      }
53      
54      /**
55       * Creates a new Location instance that is between
56       * this and <i>location</i>.
57       * 
58       * @param location the other location.
59       * @return midpoint the point between this instance and <i>location</i>.
60       */
61      public Location midpoint(Location location) {
62          double middleX = (x + location.getX()) / 2;
63          double middleY = (y + location.getY()) / 2;
64          return new Location(middleX, middleY);
65      }
66      
67      /**
68       * Transforms the coordinates of this location to the
69       * parameters supplied.
70       * 
71       * @param x the x axis value.
72       * @param y the y axis value.
73       */
74      public void transform(double x, double y) {
75          this.x = x;
76          this.y = y;
77      }
78      
79      @Override
80      public String toString() {
81          return "[" + x + ", " + y + "]";
82      }
83      
84      @Override
85      public boolean equals(Object other) {
86          if(!(other instanceof Location)) {
87              return false;
88          }
89          Location location = (Location) other;
90          return this.x == location.getX() && this.y == location.getY();
91      }
92  
93      @Override
94      public int hashCode() {
95          int hash = 7;
96          hash = 89 * hash + (int) (Double.doubleToLongBits(this.x) ^ (Double.doubleToLongBits(this.x) >>> 32));
97          hash = 89 * hash + (int) (Double.doubleToLongBits(this.y) ^ (Double.doubleToLongBits(this.y) >>> 32));
98          return hash;
99      }
100     
101     /**
102      * Creates a new location instance from a mouse event.
103      * @param event the MouseEvent instance.
104      * @return Location, the corresponding location in game.
105      */
106     public static Location toLocation(MouseEvent event) {
107         return new Location(event.getSceneX() - 320, event.getSceneY() - 240);
108     }
109     
110     /**
111      * Creates a new location, randomly.
112      * The new location cannot be near the player or out of bounds.
113      * 
114      * @return Location, a new location instance generated randomly.
115      */
116     public static Location createRandomLocation() {
117         Random random = new Random();
118         Player player = SyndromeFactory.getWorld().getPlayer();
119         int randX = random.nextInt(320);
120         int randY = random.nextInt(240);
121         if(random.nextInt(2) == 1) {
122             randX = -randX;
123         }
124         if(random.nextInt(2) == 1) {
125             randY = -randY;
126         }
127         Location result = new Location(randX, randY);
128         if(result.distanceTo(player.getLocation()) < 30) {
129             result.transform(randX + random.nextInt(25) + 15, randY + random.nextInt(25) + 15);
130         }
131         if(Math.abs(result.getX()) > 320) {
132             result.transform(randX - random.nextInt(25) + 25, randY);
133         }
134         if(Math.abs(result.getY()) > 240) {
135             result.transform(randX, randY - random.nextInt(25) + 25);
136         }
137         return result;
138     }
