23      private Stage stage;
24  
25      @Override
26      public void build(Stage stage) {
27          StackPane root = new StackPane();
28          Node background = buildBackground();
29          Scene scene = new Scene(root, 1024, 768);
30          
31          /* temporary code */
32          Rectangle r = new Rectangle();
33          r.setTranslateX(0);
34          r.setTranslateY(0);
35          r.setWidth(50);
36          r.setHeight(50);
37          
38          SyndromeFactory.getWorld().setRTest(r);
39          SyndromeFactory.getWorld().setGamePane(root);
40          SyndromeFactory.getGUIManager().setBackground(background);
41          
42          root.getChildren().add(background);
43          root.getChildren().add(r);
44          root.getChildren().add(createInfobox());
45          
46          scene.addEventHandler(MouseEvent.ANY, new MouseInput());
47          scene.addEventHandler(KeyEvent.ANY, new KeyboardInput());
48          stage.setScene(scene);
49          stage.setMaxHeight(480);
50          stage.setMaxWidth(640);
51          
52          this.stage = stage;
53      }
54

70      private Node buildBackground() {
71          Rectangle background = new Rectangle();
72          background.relocate(0, 0);
73          background.setHeight(768);
74          background.setWidth(1024);
75          background.setFill(new ImagePattern(
76                  new Image("resources/images/game_background.png")));
77          return background;
78      }
79      
80      private Group createInfobox() {
81          Group group = new Group();
82          Rectangle infobox = new Rectangle();
83          infobox.setWidth(80);
84          infobox.setHeight(60);
85          infobox.setTranslateX(270);
86          infobox.setTranslateY(-180);
87          
88          Text mainTitle = new Text("-Weapon-");
89          mainTitle.setFont(Font.font ("8BIT WONDER", 8));
90          mainTitle.setFill(Paint.valueOf("white")); 
91          mainTitle.setTranslateX(280);
92          mainTitle.setTranslateY(-170);
93          
94          Rectangle healthBar = new Rectangle();
95          healthBar.setWidth(70);
96          healthBar.setHeight(10);
97          healthBar.setFill(Paint.valueOf("green"));
98          
99          healthBar.setTranslateX(275);
100         healthBar.setTranslateY(-140);
101         
102         group.getChildren().add(infobox);
103         group.getChildren().add(mainTitle);
104         group.getChildren().add(healthBar);
105         
106         group.setTranslateX(270);
107         group.setTranslateY(-190);
108         return group;
109     }
