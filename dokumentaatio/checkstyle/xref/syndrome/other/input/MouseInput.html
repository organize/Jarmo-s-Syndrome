1   package syndrome.other.input;
2   
3   import javafx.event.EventHandler;
4   import javafx.scene.input.MouseEvent;
5   import javafx.scene.layout.Pane;
6   import syndrome.logic.projectile.Projectile;
7   import syndrome.logic.projectile.impl.Entanglement;
8   import syndrome.other.SyndromeFactory;
9   
10  public class MouseInput implements EventHandler<MouseEvent> {
11  
12      @Override
13      public void handle(MouseEvent event) {
14          if(event.getEventType().equals(MouseEvent.MOUSE_CLICKED)) {
15              handleMouseClick(event);
16          }
17      }
18      
19      private void handleMouseClick(MouseEvent event) {
20          Pane gamePane = SyndromeFactory.getWorld().getGamePane();
21          Projectile entanglement = new Entanglement();
22          gamePane.getChildren().add(entanglement.getObject());
23          entanglement.fire();
24      }
25  
26  }
