3   import javafx.event.EventHandler;
4   import javafx.scene.input.MouseEvent;
5   import syndrome.entity.Player;
6   import syndrome.logic.map.Location;
7   import syndrome.logic.projectile.Projectile;
8   import syndrome.logic.projectile.impl.Entanglement;
9   import syndrome.other.SyndromeFactory;
10  
11  /**
12   * A class that listens to mouse input when bound to a <b>Node</b>.
13   * 
14   * @note {method names are descriptive, skip doc for now?}
15   * @author Axel Wallin
16   */
17  public class MouseInput implements EventHandler<MouseEvent> {
18  
19      /**
20       * Handles all incoming mouse input.
21       * 
22       * @param event the mouse event.
23       */
24      @Override
25      public void handle(MouseEvent event) {
26          if(event.getEventType().equals(MouseEvent.MOUSE_CLICKED)) {
27              handleMouseClick(event);
28          }
29          handlePlayerRotation(event);
30      }
31      
32      private void handleMouseClick(MouseEvent event) {
33          Projectile[] multiEnt = {
34              new Entanglement(Location.toLocation(event), 90),
35              new Entanglement(Location.toLocation(event), 100),
36              new Entanglement(Location.toLocation(event), 80),
37          };
38          for(Projectile projectile : multiEnt) {
39              projectile.fire();
40              SyndromeFactory.getWorld().addProjectile(projectile, true);
41              if(SyndromeFactory.getWorld().getPlayer().getLevel() < 5) {
42                  break;
43              }
44          }
45      }
46      
47      private void handlePlayerRotation(MouseEvent e) {
48          Player player = SyndromeFactory.getWorld().getPlayer();
49          player.updateRotation(Location.toLocation(e));
50      }
51  }
