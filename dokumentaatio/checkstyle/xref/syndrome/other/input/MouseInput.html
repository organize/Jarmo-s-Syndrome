3   import javafx.event.EventHandler;
4   import javafx.scene.input.MouseEvent;
5   import javafx.scene.layout.Pane;
6   import syndrome.entity.Player;
7   import syndrome.logic.map.Location;
8   import syndrome.logic.projectile.Projectile;
9   import syndrome.logic.projectile.impl.Entanglement;
10  import syndrome.other.SyndromeFactory;
11  
12  /**
13   * A class that listens to mouse input when bound to a <b>Node</b>.
14   * 
15   * @note {method names are descriptive, skip doc for now?}
16   * @author Axel Wallin
17   */
18  public class MouseInput implements EventHandler<MouseEvent> {
19  
20      @Override
21      public void handle(MouseEvent event) {
22          if(event.getEventType().equals(MouseEvent.MOUSE_CLICKED)) {
23              handleMouseClick(event);
24          }
25          handlePlayerRotation(event);
26      }
27      
28      private void handleMouseClick(MouseEvent event) {
29          Pane gamePane = SyndromeFactory.getWorld().getGamePane();
30          Projectile entanglement = new Entanglement(Location.toLocation(event));
31          gamePane.getChildren().add(entanglement.getObject());
32          entanglement.fire();
33          SyndromeFactory.getWorld().addProjectile(entanglement);
34      }
35      
36      private void handlePlayerRotation(MouseEvent e) {
37          Player player = SyndromeFactory.getWorld().getPlayer();
38          player.updateRotation(Location.toLocation(e));
39      }
40  
41  }
