1   package syndrome;
2   
3   import java.awt.Dimension;
4   import syndrome.other.SyndromeFactory;
5   
6   /**
7    * Represents the user-defined settings and various gamestates
8    * (ie. whether the game is paused, for example).
9    * 
10   * @author Axel Wallin
11   */
12  public class GameSettings {
13      
14      private Dimension resolution;
15      private boolean paused;
16      
17      /**
18       * Creates a new instance with the default resolution of
19       * 1024x768 pixels.
20       */
21      public GameSettings() {
22          this.resolution = new Dimension(1024, 768);
23          this.paused = false;
24      }
25      
26      public Dimension getResolution() {
27          return resolution;
28      }
29      
30      /**
31       * Sets resolution with the specified dimensions.
32       * @param width the width in pixels.
33       * @param height the height in pixels.
34       */
35      public void setResolution(int width, int height) {
36          this.resolution = new Dimension(width, height);
37      }
38      
39      /**
40       * Toggles pause state across the whole game, including
41       * the timer.
42       */
43      public void togglePause() {
44          this.paused = !paused;
45          SyndromeFactory.getWorld().getTimer().setPaused(paused);
46      }
47      
48      /**
49       * Returns the current state of pause.
50       * 
51       * @return paused, whether or not the game is paused.
52       */
53      public boolean isPaused() {
54          return paused == true;
55      }
56      
57  }
