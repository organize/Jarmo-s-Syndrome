3   import javafx.application.Application;
4   import javafx.stage.Stage;
5   import syndrome.logic.map.Location;
6   import syndrome.other.SyndromeFactory;

8   /**
9    * The main class and program entry-point.
10   * 
11   * @author Axel Wallin
12   */
13  public class Syndrome extends Application {
14      
15      /**
16       * Method to set-up initial menu and audio effects.
17       * Should <i>only</i> be called from the superclass (Application).
18       * 
19       * @param primaryStage an automatically created Stage instance.
20       */
21      @Override
22      public void start(Stage primaryStage) {
23          SyndromeFactory.getAudioManager().precache("title", "title");
24          SyndromeFactory.getGUIManager().drawMainMenu();
25      }
26      
27      /**
28       * Program entry-point.
29       * Currently, the only accepted parameter is resolution (widthxheight).
30       * 
31       * @param args command-line arguments.
32       */
33      public static void main(String[] args) {
34          if(args.length != 0) {
35              String preferredResolution = args[0];
36              int width = Integer.parseInt(preferredResolution.split("x")[0]);
37              int height = Integer.parseInt(preferredResolution.split("x")[1]);
38              SyndromeFactory.getSettings().setResolution(width, height);
39          }
40          launch(args);
41      }
42      
43  }
